title = "10 ಜೋಡನೊಡೆದವಯವದ ರಕುತವ"

+++
10. ಹದನವಾದ ಬಾಣಗಳು ಕವಚಗಳನ್ನು ಸೀಳಿ ಹಾಕಿ ಅವಯವಗಳ ರಕ್ತವನ್ನು ತೋಡಿದವು. ಕೋಲ ತುದಿಯ ಚೂಪಾದ ಆಯುಧಗಳು ಒಡಲಿಗೆ ತಾಗಿ ಅವು ನೆರಿಗೆಯಂತಿರುವ ಪಟುಭಟರ ಕರುಳುಗಳನ್ನು ತಿವಿದು ಹೊರಬಂದವು. ಹೇಡಿಗಳು ಪ್ರಾಣಕ್ಕೆ ಹೆದರಿ ಬೆನ್ನು ತೋರಿಸಿ ಓಡಲು ಪ್ರಾರಂಭಿಸಿದಾಗ ಲೌಡೆಗಳು ಅವರ ನೆತ್ತಿಯ ಮೇಲೆ ಬಿದ್ದು ಅವರ ತಲೆಗಳನ್ನು ಬಿರಿಯುವಂತೆ ಮಾಡಿದವು. ಈ ಎಲ್ಲ ಅನರ್ಥಗಳು ಚತುರಂಗ ಬಲದ ಎಲ್ಲೆಡೆಯಲ್ಲೂ ನಡೆಯಿತು.
