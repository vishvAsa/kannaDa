title = "18 ತೋರದಲಿ ಕುಲಗಿರಿಯ"

+++
18. ಭಾರದಲ್ಲಿ ಮಂದರ ಪರ್ವತವನ್ನು, ಶಕ್ತಿ ಘನೀಭೂತವಾಗುವಂತೆ ವಜ್ರಾಯುಧವನ್ನು, ಶಬ್ದದಲ್ಲಿ ಮಳೆಗಾಲದ ಬರಸಿಡಿಲ ಶಕ್ತಿಯ ವಿನ್ಯಾಸವನ್ನು ತೆಗೆದು ಒಂದುಗೂಡಿಸಿ ಬ್ರಹ್ಮ ಇದನ್ನು ಸೃಷ್ಟಿ ಮಾಡಿದನೋ ಎಂಬಂತೆ ದಟ್ಟವಾದ ಭಾರಿ ರಕ್ಷಾ ಕವಚಗಳನ್ನು ಹೊಂದಿರುವ ಆನೆಯ ಸೇನೆಯ ಮಧ್ಯದಲ್ಲಿ ಕೈಯಲ್ಲಿ ಖಡ್ಗ ಹಿಡಿದಿರುವ ಭಗದತ್ತನು ಇದ್ದಾನೆ ವೀಕ್ಷಿಸು ಎಂದನು.
