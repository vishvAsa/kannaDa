title = "34 ಹೊಟ್ಟ ನೂಕಿ"

+++
34. `ನೀನು ವ್ಯರ್ಥವಾದ, ತೋರಿಕೆಯ ಯುದ್ಧದ ಜಗಜಟ್ಟಿ! ಜಳ್ಳುಗಳಂತಿರುವ ರಾಜರನ್ನು ಮುಂದೆ ನುಗ್ಗಿಸಿ ಹೊರಗೆ ಉಳಿದರೆ ನಿನ್ನನ್ನು ಬಿಡುತ್ತೇನೆಯೆ? ನಾನು ಕರ್ಣ ಎಂಬುದು ತಿಳಿದಿಲ್ಲವೆ? ನಿನ್ನನ್ನು ರಕ್ಷಿಸಿಕೊ' ಎನ್ನುತ್ತ ಕರ್ಣನು ಕಡು ಪರಾಕ್ರಮದಿಂದ ಧರ್ಮರಾಯನನ್ನು ಆಕ್ರಮಿಸಿ ಅವನು ಭಂಗಗೊಳ್ಳುವಂತೆ ಬಾಣಪ್ರಯೋಗಮಾಡುತ್ತ ಬರಲು ಸಕಲ ಮಹಾರಥರು ಅಡ್ಡಗಟ್ಟಿ ಮತ್ತೆ ಅವನನ್ನು ತಡೆದರು.
