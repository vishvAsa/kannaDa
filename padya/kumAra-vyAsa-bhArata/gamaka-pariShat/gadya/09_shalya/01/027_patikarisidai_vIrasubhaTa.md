title = "027 ಪತಿಕರಿಸಿದೈ ವೀರಸುಭಟ"

+++
27. ಶತ್ರುಗಳಾದ ಪಾಂಡವರ ಸೈನ್ಯದ ನಾಶಕ್ಕೆ ನನ್ನ ಸೇನಾಪತ್ಯದಿಂದ ನಾನು ಸಾಧನವಾಗುತ್ತೇನೆಂದು ವೀರ ಸುಭಟ ಸಮೂಹದ ಮಧ್ಯದಲ್ಲಿ ನನ್ನನ್ನು ಪ್ರೀತಿಯಿಂದ ಗೌರವಿಸಿದೆ. ಯಾವುದೇ ರೀತಿಯ ಕೃತಕ, ಮೋಸಗಳಿಲ್ಲದೇ ಧರ್ಮಜನೊಂದಿಗೆ ಯುದ್ಧ ಮಾಡುತ್ತೇನೆ. ವಿಜಯಲಕ್ಷ್ಮಿಗೆ ನೀನೆ ಪತಿಯೆನ್ನುವುದನ್ನು ತೋರಿಸುತ್ತೇನೆಂದು ಶಲ್ಯ ದುರ್ಯೋಧನನಿಗೆ ಮಾತು ಕೊಟ್ಟ.
