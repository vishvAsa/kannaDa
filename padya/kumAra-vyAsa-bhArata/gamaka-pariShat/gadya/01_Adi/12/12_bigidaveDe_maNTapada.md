title = "12 ಬಿಗಿದವೆಡೆ ಮಣ್ಟಪದ"

+++
12. ಅಲ್ಲಲ್ಲೇ ಒತ್ತೊತ್ತಾಗಿ ತೂಗು ಬಿಟ್ಟಿರುವ ಚಿನ್ನದ ಅಲಂಕಾರ ವಸ್ತುಗಳಿಂದ( ಭೋಜನಶಾಲೆ)ನೈವೇದ್ಯ ಮಂಟಪಗಳನ್ನು ಕಟ್ಟಿದರು. ಶ್ರೇಷ್ಠ ಮಣಿಗಳೊಡನೆ ನಿರ್ಮಿತವಾದ ಚಾಮರ, ಚಿನ್ನದ ಗೆಜ್ಜೆಗಳು, ಕನ್ನಡಿ, ಗಂಟೆಗಳು, ಮನೋಹರವಾದ ಬಾಸಿಂಗ - ಇವುಗಳಿಂದ ಅಲಂಕೃvವಾದÀ ಮಂಗಳ ಮಂಟಪವು ಓರಣವಾಗಿ ಕಟ್ಟಿರುವ ಬಣ್ಣದ ಸರಗಳ ಸಾಲಿನಿಂದ, ವಜ್ರಖಚಿತವಾದ ಕಂಬಗಳಿಂದ ಶೋಭಿಸಿದುವು.
