title = "23 ಆದಡಾ ಭೀಷ್ಮಾದಿ"

+++
23. ಆದರೂ ಸಹ, ಭೀಷ್ಮನೇ ಮುಂತಾದ ವೀರರು ಯುದ್ಧ ಮಾಡಿ ನೆಗ್ಗಿ ಹೋಗಿರುವ ಯುದ್ಧಭೂಮಿಯನ್ನು ಹೊಕ್ಕು ಯುದ್ಧ ಮಾಡುವ ಶಕ್ತಿಯುಳ್ಳ ಮಾದ್ರೇಶನಾದ ಶಲ್ಯನ ಹೊರತಾದ ಇತರ ವೀರರನ್ನು ನಾನು ಕಾಣೆ. ಈ ದುರಂತದ ಯುದ್ಧದಲ್ಲಿ ನಿನಗೆ ಜಯವಾದರೆ ಒಳ್ಳೆಯದು. ಆದ್ದರಿಂದ ಶಲ್ಯನಲ್ಲಿ ಸೇನಾಪತಿಯನ್ನು ಸಂಪಾದಿಸಿ - (ಶಲ್ಯನನ್ನು ಸೇನಾಪತಿನ್ನಾಗಿ ಮಾಡಿ) ಎಂದು ಕೃಪಾಚಾರ್ಯ ದುರ್ಯೋಧನನಿಗೆ ಹೇಳಿದ.
