title = "19 ಸರ್ಪಯಾಗದೊಳಾದ ಕರ್ಮದ"

+++
19. ಜನಮೇಜಯರಾಯನು ಸರ್ಪಯಾಗದಿಂದುಂಟಾದ ಅತಿಶಯವಾದ ಪಾಪಕರ್ಮಗಳನ್ನು ಕಳೆದುಕೊಂಡು , ಸುರನರರು ಬೆರಗಾಗುವಂತೆ ಉತ್ಕೃಷ್ಟವಾದ ಸೂರ್ಯನ ತೇಜಸ್ಸಿನಲ್ಲಿ ಪ್ರಕಾಶಿಸಿದನು. ಈ ಭಾರತದ ಕಥೆಯನ್ನು ಕೇಳಿದವನಿಗೆ ದೇವನಾರಿಯರ ಸಮೂಹದಲ್ಲಿ ಶೋಭಿಸುವ ಇಂದ್ರನ ಪದವಿಯು ತಪ್ಪದೇ ದೊರೆಯುತ್ತದೆ ಎಂದು ಹೇಳಿ ಸೂತನು ಕಥೆಯನ್ನು ಕೇಳುತ್ತಿದ್ದವರಿಗೆಲ್ಲರಿಗೂ ಕೈಮುಗಿದ.
