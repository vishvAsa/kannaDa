title = "023 ಮೆಚ್ಚಿದನು ಮದನಾರಿ"

+++
23. ಈಶ್ವರ ಅಶ್ವತ್ಥಾಮನನ್ನು ಮೆಚ್ಚಿಕೊಂಡ. ಹೋಮದ ಬೆಂಕಿಯು ತಾಗುವ ಮೊದಲೇ ಶಿವ ಅವನನ್ನು ತೆಗೆದು ತನ್ನ ಬಿಚ್ಚಿದ ಜಡೆಗಳ ನಡುವೆ ಇದ್ದ ಗಂಗಾಜಲದಲ್ಲಿ ವೀರಭಟನಾದ ಅಶ್ವತ್ಥಾಮನನ್ನು ನೆನೆಸಿದನು. ನೀನು ನನಗೆ ಹೊಡೆದ ಬಾಣಗಳಿವೆ, ಖಡ್ಗವಿದೆ. ಇವುಗಳನ್ನು ತೆಗೆದುಕೋ  ಇನ್ನು ಮುಚ್ಚುಮರೆಯೇ (ಭೂತದ ರೂಪದಲ್ಲಿರುವ ಶಿವನೆಂಬ ವಿಚಾರದಲ್ಲಿ) ನನ್ನ ಮಕ್ಕಳಲ್ಲಿ ನೀನು ಹಿರಿಯವ - ಎಂದು ಶಿವ, ಅಶ್ವತ್ಥಾಮನ ಮೈದಡವಿದ.
