title = "17 ಹಿಡಿದು ರಾಜ್ಯವ"

+++
17. ಉತ್ತರನ ಪೌರುಷದ ಮಾತುಗಳು ಮುಂದುವರಿದಿವೆ.   
"ಅಲ್ಲ ! ಧರ್ಮರಾಯನನ್ನು ಹಿಡಿದು ಜೂಜಿನಲ್ಲಿ ರಾಜ್ಯವನ್ನು ಕಿತ್ತುಕೊಂಡು ಪಾಪ ಹೆಂಗಸಾದ ಆ ದ್ರೌಪದಿಗೆ ಅವಮಾನ ಮಾಡಿ ಪಾಂಡವರನ್ನು ರಾಜ್ಯದಿಂದ ಓಡಿಸಿ ಕೊಬ್ಬಿದ ತನ್ನ ಸಾಹಸವನ್ನು ನನ್ನ ಮೇಲೆ ತೋರಿಸುತ್ತಿದ್ದಾನೇನು ಆ ಕೌರವ ? ನನ್ನನ್ನು ಪಾಪ ಬಡಪಾಯಿ À ಧರ್ಮರಾಯ ಎಂದು ಭಾವಿಸಿದ್ದಾನೆಯೆ ? ನಾನು ಕೆರಳಿ ನಿಂತರೆ ಕೌರವನ ಕೀರ್ತಿಯನ್ನೇ ಅಳಿಸಿಹಾಕಬಲ್ಲೆ. ಪಾಪ ಇದು ಕೌರವನಿಗೆ ತಿಳಿಯದು ಎಂದು ಆ ಸುಕುಮಾರ ಸಿಟ್ಟಿನಿಂದ ನುಡಿದ.
