title = "023 ಕಣ್ಡುದಿವರನು ರಾಯ"

+++
23. "ಇವರನ್ನು ಕಂಡು ರಾಯನ ಸೇನೆ ಓಡಿ ಹೋಯಿತು. ದೊರೆಗಳು ಇವರು.  ಇದು ಬಿಲ್ವಿದ್ಯೆಯಲ್ಲಿ ಕೋದಂಡವನ್ನು ಹೊಂದಿರುವ ರುದ್ರನ ಯುದ್ಧವಲ್ಲವೇ?  ನಮ್ಮ ಕಡೆಯವರು ಏನು ಮಾಡಲು ಸಾಧ್ಯ ? ಸಾತ್ಯಕಿಯು  ಏಟುಗಳನ್ನು ತಿಂದಿದ್ದರೂ ಸಹ ಅದನ್ನು ಮೀರಿನಿಂತ ಅಥವಾ ಅದಕ್ಕೆ ಔಷಧಿಯ ರೂಪದ ಶೂರನೋ ಅಥವಾ ದ್ರೋಣಾಸ್ತ್ರದ ರುಚಿಯನ್ನು ನೋಡಿದ ಶೂರನೋ" ಎನ್ನುತ್ತಾ ಭೀಮನು ಗರ್ಜಿಸಿದನು.
