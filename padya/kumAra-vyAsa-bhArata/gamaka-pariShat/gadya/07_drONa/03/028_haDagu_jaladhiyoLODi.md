title = "028 ಹಡಗು ಜಲಧಿಯೊಳೋಡಿ"

+++
28. ಸಮುದ್ರದಲ್ಲಿ ಹಡಗು ಓಡುವಾಗ ಸಮುದ್ರದೊಳಗಿರುವ ಪರ್ವತಕ್ಕೆ ಡಿಕ್ಕಿ ಹೊಡೆದು ನುಗ್ಗಾಗುವಂತೆ ಈ ಆನೆಯ ಮೇಲೆ ಬಿದ್ದು ಅದಕ್ಕೆ ತಾಗಿದ ಸೈನಿಕರೇ ನೋವನ್ನನುಭವಿಸಿದರು. ದೇಹದ ಮೇಲೆ ಸ್ವಲ್ಪವೂ ಅಭಿಮಾನವನ್ನಿಟ್ಟುಕೊಳ್ಳದೆ ಆನೆಯ ಮೇಲೆ ಯುದ್ಧವನ್ನು ಮಾಡಿದರು. ದೋಷಿಯಾದವನಿಗೆ ಹಾವನ್ನು ಒಳಗೆ ಉಳ್ಳ ಕೊಡದೊಳಗೆ ಕೈಹಾಕುವುದು ಸುಲಭವೇ ಕೇಳು ಧೃತರಾಷ್ಟ್ರ ಎಂದು ಸಂಜಯ ಹೇಳಿದ.
